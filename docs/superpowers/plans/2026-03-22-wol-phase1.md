> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add MAC address validation with normalization, auto-detect the NAS's own MAC address, and show it as a suggestion in the frontend config panel.

**Architecture:** A shared MAC validator function in `backend/app/schemas/validators.py` is used by `WolRequest` and `SleepConfigUpdate`. The `SleepBackend` ABC gets a new `get_own_mac()` method, implemented in both Linux and Dev backends. The frontend shows the detected MAC as a clickable "Übernehmen" suggestion.

**Tech Stack:** Python/Pydantic (validator), Linux `/proc/net/route` + `/sys/class/net/` (MAC detection), React/TypeScript (frontend)

**Spec:** `docs/superpowers/specs/2026-03-22-wol-completion-design.md` — Phase 1

```python
"""Shared Pydantic validators reusable across schemas."""
import re
from typing import Optional

_MAC_RE = re.compile(
    r"^([0-9A-Fa-f]{2})[:\-]"
    r"([0-9A-Fa-f]{2})[:\-]"
    r"([0-9A-Fa-f]{2})[:\-]"
    r"([0-9A-Fa-f]{2})[:\-]"
    r"([0-9A-Fa-f]{2})[:\-]"
    r"([0-9A-Fa-f]{2})$"
)


def validate_mac_address(value: Optional[str]) -> Optional[str]:
    """Validate and normalize a MAC address.

    - None → None (field not updated)
    - "" → None (field cleared)
    - Valid MAC → uppercase colon-separated (AA:BB:CC:DD:EE:FF)
    - Invalid → raises ValueError
    """
    if value is None:
        return None
    value = value.strip()
    if not value:
        return None
    match = _MAC_RE.match(value)
    if not match:
        raise ValueError(
            f"Invalid MAC address format: '{value}'. "
            "Expected AA:BB:CC:DD:EE:FF or AA-BB-CC-DD-EE-FF"
        )
    return ":".join(g.upper() for g in match.groups())
```

```python
    async def get_own_mac(self) -> Optional[str]:
        """Detect MAC of the primary NIC via /proc/net/route."""
        return await self._get_own_mac_from_paths(
            "/proc/net/route", "/sys/class/net"
        )

    async def _get_own_mac_from_paths(
        self, route_path: str, net_dir: str
    ) -> Optional[str]:
        """Testable implementation: reads route and sysfs from given paths."""
        try:
            with open(route_path) as f:
                for line in f:
                    parts = line.strip().split()
                    # Find the default route: Destination == 00000000
                    if len(parts) >= 2 and parts[1] == "00000000":
                        iface = parts[0]
                        mac_path = f"{net_dir}/{iface}/address"
                        try:
                            with open(mac_path) as mf:
                                raw = mf.read().strip()
                            if raw and raw != "00:00:00:00:00:00":
                                return raw.upper()
                        except OSError:
                            logger.warning("Cannot read MAC from %s", mac_path)
                            return None
        except OSError:
            logger.debug("Cannot read route table from %s", route_path)
        return None
```
